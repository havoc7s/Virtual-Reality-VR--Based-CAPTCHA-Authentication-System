using UnityEngine;
using TMPro;
using UnityEngine.SceneManagement;

public class UIManager : MonoBehaviour
{
    [Header("UI Elements")]
    public TextMeshProUGUI timerText;
    public TextMeshProUGUI attemptsText;

    [Header("Settings")]
    public float totalTime = 30f;
    public int maxAttempts = 3;
    public float minSolveTime = 3f;

    private float timeRemaining;
    private int attemptsRemaining;
    private bool isRunning = true;
    private float sessionStartTime;

    public static UIManager Instance;

    private void Awake()
    {
        if (Instance == null)
            Instance = this;
        else
            Destroy(gameObject);
    }

    private void Start()
    {
        sessionStartTime = Time.time;
        timeRemaining = totalTime;

        // Reset wrong placements each session
        PieceInteractable.TotalWrongPlacements = 0;

        // Read remaining retries
        if (PlayerPrefs.HasKey("AttemptsRemaining"))
            attemptsRemaining = PlayerPrefs.GetInt("AttemptsRemaining");
        else
            attemptsRemaining = maxAttempts;

        UpdateUI();
    }

    private void Update()
    {
        if (!isRunning) return;

        timeRemaining -= Time.deltaTime;

        if (timeRemaining <= 0)
        {
            timeRemaining = 0;
            isRunning = false;
            OnTimeUp();
        }

        UpdateUI();
    }

    private void UpdateUI()
    {
        timerText.text =
            $"Time: {Mathf.CeilToInt(timeRemaining)}";

        attemptsText.text =
            $"Retries Left: {attemptsRemaining}";

        if (timeRemaining <= 10f)
            timerText.color = Color.red;
        else if (timeRemaining <= 20f)
            timerText.color = Color.yellow;
        else
            timerText.color = Color.white;
    }

    // ==========================================
    // CAPTCHA PASSED
    // ==========================================
    public void OnCaptchaSolved()
    {
        float solveTime =
            Time.time - sessionStartTime;

        // BOT DETECTION
        if (solveTime < minSolveTime)
        {
            OnFailed("BOT DETECTED!");
            return;
        }

        isRunning = false;

        PlayerPrefs.DeleteKey("AttemptsRemaining");
        PlayerPrefs.Save();

        timerText.color = Color.green;
        timerText.text = "PASSED!";
        attemptsText.text = "Verified";

        Debug.Log(
            $"=== CAPTCHA PASSED in {solveTime:F2}s ==="
        );

        int retryAttempts =
            maxAttempts - attemptsRemaining;

        int wrongPlacements =
            PieceInteractable.TotalWrongPlacements;

        if (BackendManager.Instance != null)
        {
            StartCoroutine(
                BackendManager.Instance.SubmitResult(
                    solveTime,
                    retryAttempts,
                    wrongPlacements
                )
            );
        }

        Invoke("LoadMainVR", 2f);
    }

    // ==========================================
    // TIME UP
    // ==========================================
    private void OnTimeUp()
    {
        Debug.Log("=== TIME UP! ===");
        UseAttempt();
    }

    // ==========================================
    // USE RETRY
    // ==========================================
    public void UseAttempt()
    {
        attemptsRemaining--;

        PlayerPrefs.SetInt(
            "AttemptsRemaining",
            attemptsRemaining
        );

        PlayerPrefs.Save();

        if (attemptsRemaining <= 0)
        {
            OnFailed("FAILED!");
            return;
        }

        timerText.text =
            $"Restarting... ({attemptsRemaining} left)";

        timerText.color = Color.yellow;

        Invoke("RestartCaptcha", 2f);
    }

    private void RestartCaptcha()
    {
        SceneManager.LoadScene(
            SceneManager.GetActiveScene().name
        );
    }

    // ==========================================
    // FAILED
    // ==========================================
    private void OnFailed(string message = "FAILED!")
    {
        isRunning = false;

        PlayerPrefs.DeleteAll();
        PlayerPrefs.Save();

        timerText.color = Color.red;
        timerText.text = message;
        attemptsText.text = "Access Denied";

        Debug.Log($"=== {message} ===");

        float elapsed =
            totalTime - timeRemaining;

        int retryAttempts =
            maxAttempts - attemptsRemaining;

        int wrongPlacements =
            PieceInteractable.TotalWrongPlacements;

        if (BackendManager.Instance != null)
        {
            StartCoroutine(
                BackendManager.Instance.SubmitResult(
                    elapsed,
                    retryAttempts,
                    wrongPlacements
                )
            );
        }

        Invoke("LoadAccessDenied", 2f);
    }

    private void LoadMainVR()
    {
        SceneManager.LoadScene("MainVRScene");
    }

    private void LoadAccessDenied()
    {
        SceneManager.LoadScene("AccessDeniedScene");
    }
}
